Quick start

// 创建一个日期对象
var date = new Date();

// 默认输出：
date -> Fri Aug 26 2016 10:45:58 GMT+0800 (中国标准时间)

// 格式化为时间字符串输出：
date.format() -> "2016-08-26 10:45:58 812"

// 获取本季度的天数：
date.getQuarterDays() -> 92

// 设置日期的时间为零点 00:00:00 000，返回毫秒数：
date.setTimeToFirst() -> 1472140800000

// 零点时间格式化为时间字符串输出：
date.format() -> "2016-08-26 00:00:00 000"
