Quick start

// 创建一个日期对象
var date = new Date();

// 是否为闰年：
date.isLeapYear() -> true

// 获取本季度的天数：
date.getQuarterDays() -> 92

// 格式化为时间字符串：
date.format() -> "2016-08-31 00:04:12 891"

// 格式化为指定格式的时间字符串：
date.format(Date.FORMAT_DATE) -> "2016-08-31"

// 设置为当天的初始时间（0点）
date.setTimeToFirst() -> 1472572800000

// 格式化时间
date.format() -> "2016-08-31 00:00:00 00"
